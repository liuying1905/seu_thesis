\section{公式}

需要等号对齐的公式写法。

\begin{equation}
\begin{split}
\nabla_{\theta}J(\pi_{\theta})
& = \int_{\mathcal{S}}\rho^{\pi}(s)\int_{\mathcal{A}} \nabla_{\theta}\pi_{\theta}(a|s)Q^{\pi}(s,a)dads \\
& = \mathbb{E}_{s\sim\rho^{\pi},a\sim\pi_{\theta}}[\nabla
_{\theta}\log\pi_{\theta}(a|s)Q^{\pi}(s,a)]
\end{split}
\end{equation}
